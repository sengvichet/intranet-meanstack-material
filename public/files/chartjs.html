ChartJS Charts

Chart.js Simple yet flexible JavaScript charting for designers & developers.

Area Chart


<canvas class="chart chart-line" chart-data="area.data" chart-labels="area.labels" chart-series="area.series" chart-options="area.options" chart-dataset-override="area.datasetOverride" chart-colors="area.colors" height="110"></canvas>
                            

$scope.area = {
    labels: ["January", "February", "March", "April", "May", "June", "July"],
    series: ['Series A', 'Series B'],
    data: [
        [65, 59, 80, 81, 56, 55, 40],
        [28, 48, 40, 19, 86, 27, 90]
    ],
    datasetOverride: [{ yAxisID: 'y-axis-1' }, { yAxisID: 'y-axis-2' }],
    colors: ['#1A237E', '#C62828'],
    options: {
        title: {
            display: true,
            text: 'Visitors Area Chart'
        },
        scales: {
            yAxes: [{
                    id: 'y-axis-1',
                    type: 'linear',
                    display: true,
                    position: 'left'
                },
                {
                    id: 'y-axis-2',
                    type: 'linear',
                    display: true,
                    position: 'right'
                }
            ]
        },
        legend: { display: true }
    }
};

Mixed type Chart


 <canvas class="chart chart-bar" chart-data="mixedtype.data" chart-labels="mixedtype.labels" chart-series="mixedtype.series" chart-colors="mixedtype.colors" chart-options="mixedtype.options" chart-dataset-override="mixedtype.datasetOverride" height="110"></canvas>
                            

 $scope.mixedtype = {
    labels: ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'],
    series: ['Series A', 'Series B'],
    data: [
        [65, -59, 80, 81, -56, 55, -40],
        [28, 48, -40, 19, 86, 27, 90]
    ],
    colors: ['#1A237E', '#448AFF'],
    options: {
        title: {
            display: true,
            text: 'Visitors Mixed Type Chart'
        },
        legend: { display: true }
    },
    datasetOverride: [{
            label: "Bar chart",
            borderWidth: 1,
            type: 'bar'
        },
        {
            label: "Line chart",
            borderWidth: 3,
            hoverBackgroundColor: "rgba(255,99,132,0.4)",
            hoverBorderColor: "rgba(255,99,132,1)",
            type: 'line'
        }
    ]
};
