Select

Select allows user input through specified options. The <md-select> component can be used within a <md-input-container> or as a stand alone component by using the md-no-underline class.

    user: null,
    users: null
};

$scope.asyncSearchSelectloadUsers = function() {

    // Use timeout to simulate a 650ms request.
    return $timeout(function() {

        $scope.asyncSearchSelect.users = $scope.asyncSearchSelect.users || [
            { id: 1, name: 'Scooby Doo' },
            { id: 2, name: 'Shaggy Rodgers' },
            { id: 3, name: 'Fred Jones' },
            { id: 4, name: 'Daphne Blake' },
            { id: 5, name: 'Velma Dinkley' }
        ];

    }, 650);
};

$scope.selectTextDemo = {
    items: [1, 2, 3, 4, 5, 6, 7],
    getSelectedText: function() {
        if ($scope.selectTextItem.data) {
            return "You have selected: Item " + $scope.selectTextItem.data;
        } else {
            return "Please select an item";
        }
    }
};
